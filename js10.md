# Create the below patterns using JavaScript
# 1.	Square star pattern
<img src="square star.png">

```
let n = 5;
let str = "";

for (let i = 1; i <= n; i++) {
    for (let j = 1; j <= n; j++) {
        str += "*";
    }
    str += "\n";
}

console.log(str);

```

# 4.	Right angle pattern
<img src="rightangle.png">

```
str = "";
for (let i = 1; i <= n; i++) {
    for (let j = 1; j <= n; j++) {
        if (j > n - i) {
            str += "*";
        } else {
            str += " ";
        }
    }
    str += "\n";
}

console.log(str);

```

# 6.	Pyramid
<img src="pyramid.png">

```
str = "";
for (let i = 1; i <= n; i++) {
    for (let j = 1; j <= n - i; j++) {
        str += " ";
    }
    for (let k = 1; k <= 2 * i - 1; k++) {
        str += "*";
    }
    str += "\n";
}

console.log(str);

```

# 10. hollow square pattern
<img src="hollowsq.png">

```
str = "";
for (let i = 1; i <= n; i++) {
    for (let j = 1; j <= n; j++) {
        if (i === 1 || i === n || j === 1 || j === n) {
            str += "*";
        } else {
            str += " ";
        }
    }
    str += "\n";
}
```
